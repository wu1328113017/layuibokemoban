js代码的执行顺序可以在外面先声明一个Boolean值为false，在需要先执行完的代码后改变这个Boolean值为true，然后在需要后面执行的代码加个定时器setInterval(callback,speed),这个定时器的作用是每speed时间执行一次callback函数，然后我们把后面要执行的代码放在callback里面，加个if判断，判断设置的这个Boolean值是否为true，即先执行的代码是否执行完毕，如果为true，则改变该Boolean为false(这个可有可无,主要是为了不必要的情况出现防止重复执行),同时后执行的代码执行完后使用 clearInterval()函数清除定时器，防止一直循环造成内存浪费，代码示例如下:

                         var bln = false;

                         var timer = setInterval(function(){

                               if(bln == true){

                                  bln = false;//防止意外造成死循环

                                  //这里放入你后执行的代码

                                  clearInterval(timer);//清除定时器

                                 }

                           },500);//这个500可任意设置

                           //这里放入你要先执行的代码

                           //执行完后改变bln为true

                            bln = true;
